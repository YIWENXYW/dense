149  void calc_neighbor_2d() {
150  for (dense::Natural i = 0; i < _cells_total; i++) {
151  int adjacents[6];
152 
153  /* Hexagonal Adjacencies
154  0: TOP
155  1: TOP-RIGHT
156  2: BOTTOM-RIGHT
157  3: BOTTOM
158  4: TOP-LEFT
159  5: BOTTOM-LEFT
160  */
161  if (i % 2 == 0) {
162  adjacents[0] = (_cells_total + i - _width_total) % _cells_total;
163  adjacents[1] = (_cells_total + i - _width_total + 1) % _cells_total;
164  adjacents[2] = (i + 1) % _cells_total;
165  adjacents[3] = (i + _width_total) % _cells_total;
166  if (i % _width_total == 0) {
167  adjacents[4] = i + _width_total - 1;
168  adjacents[5] = (_cells_total + i - 1) % _cells_total;
169  } else {
170  adjacents[4] = i - 1;
171  adjacents[5] = (_cells_total + i - _width_total - 1) % _cells_total;
172  }
173  } else {
174  adjacents[0] = (_cells_total + i - _width_total) % _cells_total;
175  if (i % _width_total == _width_total - 1) {
176  adjacents[1] = i - _width_total + 1;
177  adjacents[2] = (i + 1) % _cells_total;
178  } else {
179  adjacents[1] = i + 1;
180  adjacents[2] = (_cells_total + i + _width_total + 1) % _cells_total;
181  }
182  adjacents[3] = (i + _width_total) % _cells_total;
183  adjacents[4] = (i + _width_total - 1) % _cells_total;
184  adjacents[5] = (_cells_total + i - 1) % _cells_total;
185  }
