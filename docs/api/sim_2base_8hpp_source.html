153  * CALC_NEIGHBOR_2D
154  * populates the data structure "_neighbors" with cell indices of neighbors
155  * follows hexagonal adjacencies for an unfilled tube
156  */
157  IF_CUDA(__host__ __device__)
158  void calc_neighbor_2d(){
159  for (dense::Natural i = 0; i < _cells_total; i++) {
160  int adjacents[6];
161 
162  /* Hexagonal Adjacencies
163  0: TOP
164  1: TOP-RIGHT
165  2: BOTTOM-RIGHT
166  3: BOTTOM
167  4: TOP-LEFT
168  5: BOTTOM-LEFT
169  */
170  if (i % 2 == 0) {
171  adjacents[0] = (_cells_total + i - circumf) % _cells_total;
172  adjacents[1] = (_cells_total + i - circumf + 1) % _cells_total;
173  adjacents[2] = (i + 1) % _cells_total;
174  adjacents[3] = (i + circumf) % _cells_total;
175  if (i % circumf == 0) {
176  adjacents[4] = i + circumf - 1;
177  adjacents[5] = (_cells_total + i - 1) % _cells_total;
178  } else {
179  adjacents[4] = i - 1;
180  adjacents[5] = (_cells_total + i - circumf - 1) % _cells_total;
181  }
182  } else {
183  adjacents[0] = (_cells_total + i - circumf) % _cells_total;
184  if (i % circumf == circumf - 1) {
185  adjacents[1] = i - circumf + 1;
186  adjacents[2] = (i + 1) % _cells_total;
187  } else {
188  adjacents[1] = i + 1;
189  adjacents[2] = (_cells_total + i + circumf + 1) % _cells_total;
190  }
191  adjacents[3] = (i + circumf) % _cells_total;
192  adjacents[4] = (i + circumf - 1) % _cells_total;
193  adjacents[5] = (_cells_total + i - 1) % _cells_total;
194  }
195 
196  if (i % circumf == 0) {
197  _neighbors[i][0] = adjacents[0];
198  _neighbors[i][1] = adjacents[1];
199  _neighbors[i][2] = adjacents[4];
200  _neighbors[i][3] = adjacents[5];
201  _numNeighbors[i] = 4;
202  } else if ((i + 1) % circumf == 0) {
203  _neighbors[i][0] = adjacents[0];
204  _neighbors[i][1] = adjacents[1];
205  _neighbors[i][2] = adjacents[2];
